let currentPage = 'home';

function goTo(page) {
  // Hide all pages
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.getElementById('page-' + page).classList.add('active');

  // Update nav active state
  document.querySelectorAll('.nav-links a').forEach(a => a.classList.remove('active-link'));
  const navEl = document.getElementById('nav-' + page);
  if (navEl) navEl.classList.add('active-link');

  currentPage = page;
  window.scrollTo({ top: 0, behavior: 'smooth' });
  setTimeout(initReveal, 80);
}

  mobileOpen = !mobileOpen;

}

  document.getElementById('modalSpecs').innerHTML = p.specs.map(s => `<div class="modal-spec">${s}</div>`).join('');
  document.getElementById('modalBackdrop').classList.add('open');
  document.body.style.overflow = 'hidden';
}

function closeModal() {
  document.getElementById('modalBackdrop').classList.remove('open');
  document.body.style.overflow = '';
}

function closeModalOnBackdrop(e) {
  if (e.target === document.getElementById('modalBackdrop')) closeModal();
}

function toggleFeature(idx) {
  const item = document.getElementById('fi-' + idx);
  const isOpen = item.classList.contains('open');
  document.querySelectorAll('.feature-item').forEach(i => i.classList.remove('open'));
  if (!isOpen) item.classList.add('open');
}

let toastTimer;
function showToast(msg) {
  const t = document.getElementById('toast');
  document.getElementById('toastMsg').textContent = msg;
  t.classList.add('show');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => t.classList.remove('show'), 3500);
}

function handleFormSubmit() {
  const fields = ['f-name','f-tel','f-email','f-product','f-msg'];
  const allFilled = fields.every(id => document.getElementById(id).value.trim() !== '');
  if (!allFilled) { showToast('⚠️ กรุณากรอกข้อมูลให้ครบทุกช่อง'); return; }
  showToast('✅ ส่งข้อความเรียบร้อย ทีมงานจะติดต่อกลับภายใน 24 ชม.');
  fields.forEach(id => { document.getElementById(id).value = ''; });
}

/* ================================================================
   SCROLL REVEAL
   ================================================================ */
function initReveal() {
  const page = document.getElementById('page-' + currentPage);
  if (!page) return;
  const els = page.querySelectorAll('.reveal');
  const io = new IntersectionObserver((entries) => {
    entries.forEach((e, i) => {
      if (e.isIntersecting) {
        setTimeout(() => e.target.classList.add('visible'), i * 70);
        io.unobserve(e.target);
      }
    });
  }, { threshold: 0.06 });
  els.forEach(el => { el.classList.remove('visible'); io.observe(el); });
}
